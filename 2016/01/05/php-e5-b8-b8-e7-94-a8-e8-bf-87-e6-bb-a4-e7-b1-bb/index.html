      { $count = IString::getStrLen($str); if($count > $length)

      } return $str;

    • @brief 对字符串进行过滤处理

    • @return string 被过滤后的字符串
    • @note 默认执行的是string类型的过滤
      */ public static function act($str,$type = ‘string’,$limitLen = false) { if(is_array($str))
      { $resultStr = array(); foreach($str as $key => $val)
      { $key = self::act($key, $type, $limitLen); $val = self::act($val, $type, $limitLen); $resultStr\[$key\] = $val;
} return $resultStr;
      } else { switch($type)
      { case "int": return intval($str); break; case "float": return floatval($str); break; case "text": return self::text($str,$limitLen); break; case "bool": return (bool)$str; break; case "url": return self::clearUrl($str); break; case "filename": return self::fileName($str); break; // 严格过滤字符串 case "strict": return self::strict($str); // 检测邮箱地址 case "email": return self::email(); // 检测http网址 case "http": return self::http(); default: return self::string($str,$limitLen); break;
