  \* @return \[type\]      \[description\]
  */
function utf8_str($str){  
    $mb = mb_strlen($str,'utf-8');  
    $st = strlen($str);  
    if($st==$mb){
        return FALSE;  
    }elseif($st%$mb==0 && $st%3==0){
        return TRUE;
    }else{
        return FALSE;
    }
}

/**
* 防止xss攻击
* @blog http://www.phpddt.com
* @param $string
* @param $low 安全别级低
*/
function clean_xss(&$string, $low = False)
{
if (! is_array ( $string ))
{
$string = trim ( $string );
$string = strip_tags ( $string );
$string = htmlspecialchars ( $string );
if ($low)
{
return True;
}
$string = str_replace ( array (‘“‘, “\\”, “‘“, “/“, “..”, “../“, “./“, “//“ ), ‘’, $string );
$no = ‘/%0[0-8bcef]/‘;
$string = preg_replace ( $no, ‘’, $string );
$no = ‘/%1[0-9a-f]/‘;
$string = preg_replace ( $no, ‘’, $string );
$no = ‘/[\x00-\x08\x0B\x0C\x0E-\x1F\x7F]+/S’;
$string = preg_replace ( $no, ‘’, $string );
return True;
}
$keys = array_keys ( $string );
foreach ( $keys as $key )
{
clean_xss ( $string [$key] );
}
}

/\*\*
 \* \[过滤字符防止sql注入\]
 \* @param  \[type\] $post \[description\]
 \* @return \[type\]       \[description\]
 */
public function safe($post){
    $post = trim($post);
    if (!get\_magic\_quotes\_gpc()) { // 判断magic\_quotes_gpc是否为打开   
        $post = addslashes($post); // 进行magic\_quotes\_gpc没有打开的情况对提交数据的过滤   
    }   
    $post = str\_replace("\_", "oo", $post); // 把 '_'过滤掉   
    $post = str_replace("%", "oo", $post); // 把 '%'过滤掉   
    $post = nl2br($post); // 回车转换   
    $post = htmlspecialchars($post); // html标记转换 
    $post = stripslashes($post); // html标记转换 

    return $post;
}
